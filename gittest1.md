* ## (h6) __코드 나타내기__
    * for문으로 별피라미드 찍기

 #include <stdio.h>

int main(){

     int i, j, n;

    scanf_s("%d", &n);

    for(i=0; i<n; i++){

      for(j=i; j<(n-1); j++){
          
         printf(" ");
      }
      for(j=0; j<=(i*2); j++){

         printf("*");
      }
      printf("\n");
     }

   return 0;

}
